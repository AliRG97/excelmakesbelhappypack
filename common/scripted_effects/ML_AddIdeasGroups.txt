ML_on_country_released = {
	log = test.4
	FROM = {
		# save the number of ideas in variables
		GetAllIdeas = yes
		if = { 	limit = { check_variable = {
			which = admin
			value = 1
		}}
			PREV = {

				set_variable = {
					which = admin
					which = PREV
				}
				log = test.5
				#add_idea_group = administrative_ideas
				while = { #loop
					limit = { check_variable = {
						which = admin
						value = 1
						} }
					random_list = { # this might not be a seeded scope and evry loop wil give the same option
						30 = {
							Addinnovativeness_ideas  = yes
						}
						30 = {
							Addeconomic_ideas = yes
						}
					}
				} 
				if = {
					limit = {
						check_variable = {
							which = admin
							value = 1
							}
					}
					log = ErrorText.4
				}
			}
		}
	}

}
