#calc if provence $first$ has lower or equal cost to dev than $second$ 
is_dev_cost_lower = {
	variable_arithmetic_trigger = {
		export_to_variable = {
			variable_name = LocalDeVCost1
			value = modifier:local_development_cost
			who = $first$ 
		}
		export_to_variable = {
			variable_name = LocalDeVCost2
			value = modifier:local_development_cost
			who = $second$ 
		}
		NOT = {	check_variable = {
			which = LocalDeVCost1
			which = LocalDeVCost2
		} }
	}
}
